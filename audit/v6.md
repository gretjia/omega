def calc_winding_number(x_trace: np.ndarray, y_trace: np.ndarray) -> float:
    """
    v6.0 Topology: Winding Number (Auditor P0: 4.1).
    Measures topological loops (cycles) in Phase Space.
    """
    if x_trace.size < 3: return 0.0
    
    # Center coordinates
    x_c = x_trace - np.mean(x_trace)
    y_c = y_trace - np.mean(y_trace)
    
    # Calculate phase angles
    angles = np.arctan2(y_c, x_c)
    
    # Unwrap phase (detect +/- 2pi jumps)
    deltas = np.diff(angles)
    deltas = np.where(deltas > np.pi, deltas - 2 * np.pi, deltas)
    deltas = np.where(deltas < -np.pi, deltas + 2 * np.pi, deltas)
    
    total_rotation = np.sum(deltas)
    return float(total_rotation / (2 * np.pi))

def _rename_map(cfg: L2PipelineConfig) -> Dict[str, str]:
    m = cfg.mapping
    rename = {
        m.col_symbol: "symbol", m.col_exchange: "exchange",
        m.col_date: "date", m.col_time: "time",
        m.col_price: "price", m.col_volume: "vol",
        m.col_turnover: "turnover", m.col_bs_flag: "bs_flag",
    }
    for i in range(1, int(m.book_levels) + 1):
        rename[f"{m.bid_price_prefix}{i}"] = f"bid_p{i}"
        rename[f"{m.bid_volume_prefix}{i}"] = f"bid_v{i}"
        rename[f"{m.ask_price_prefix}{i}"] = f"ask_p{i}"
        rename[f"{m.ask_volume_prefix}{i}"] = f"ask_v{i}"
    return rename

def _sum_expr(exprs: List[pl.Expr]) -> pl.Expr:
    return pl.fold(pl.lit(0.0), lambda acc, x: acc + x, exprs)

def _numeric_cols(cfg: L2PipelineConfig) -> List[str]:
    m = cfg.mapping
    cols = ["price", "vol", "turnover"]
    for i in range(1, int(m.book_levels) + 1):
        cols.extend([f"bid_p{i}", f"bid_v{i}", f"ask_p{i}", f"ask_v{i}"])
    return cols

def scan_l2_quotes(path: str, cfg: L2PipelineConfig) -> pl.LazyFrame:
    is_parquet = path.lower().endswith(".parquet")
    if is_parquet or cfg.io.input_format == "parquet":
        lf = pl.scan_parquet(path)
    elif cfg.io.input_format == "csv":
        try:
            lf = pl.read_csv(path, encoding=cfg.io.csv_encoding, infer_schema_length=0).lazy()
        except Exception:
            if is_parquet: return pl.scan_parquet(path)
            raise
    else:
        if is_parquet: lf = pl.scan_parquet(path)
        else: raise ValueError(f"Unknown format: {cfg.io.input_format}")
            
    rename = _rename_map(cfg)
    schema = lf.collect_schema()
    valid_rename = {k: v for k, v in rename.items() if k in schema.names()}
    if not valid_rename and "microprice" in schema.names(): return lf
    
    lf = lf.select([pl.col(c) for c in valid_rename.keys()]).rename(valid_rename)
    num_cols = _numeric_cols(cfg)
    existing = lf.collect_schema().names()
    cols_to_cast = [c for c in num_cols if c in existing]
    
    lf = lf.with_columns(
        [pl.col("time").cast(pl.Int64, strict=False)] + 
        [pl.col(c).cast(pl.Float64, strict=False) for c in cols_to_cast]
    )
    return lf

def _microprice_expr(cfg: L2PipelineConfig) -> pl.Expr:
    depth = int(cfg.microprice.depth_level)
    bid_terms = [pl.col(f"bid_p{i}") * pl.col(f"ask_v{i}") for i in range(1, depth + 1)]
    ask_terms = [pl.col(f"ask_p{i}") * pl.col(f"bid_v{i}") for i in range(1, depth + 1)]
    denom_terms = [pl.col(f"bid_v{i}") + pl.col(f"ask_v{i}") for i in range(1, depth + 1)]
    return (_sum_expr(bid_terms + ask_terms) / _sum_expr(denom_terms).clip(lower_bound=1e-9)).alias("microprice")

def _depth_expr(cfg: L2PipelineConfig) -> pl.Expr:
    depth = int(max(cfg.microprice.depth_level, cfg.ofi.depth_levels))
    bid_v = [pl.col(f"bid_v{i}") for i in range(1, depth + 1)]
    ask_v = [pl.col(f"ask_v{i}") for i in range(1, depth + 1)]
    return (_sum_expr(bid_v + ask_v) / (2.0 * depth)).alias("depth")

def _ofi_expr(cfg: L2PipelineConfig) -> pl.Expr:
    ofi_cfg = cfg.ofi
    if ofi_cfg.ofi_mode == "price_change":
        return (pl.when(pl.col("price") > pl.col("price").shift(1)).then(pl.col("vol_tick"))
                .when(pl.col("price") < pl.col("price").shift(1)).then(-pl.col("vol_tick"))
                .otherwise(float(ofi_cfg.ofi_zero_value)).alias("v_ofi")
        )
    # Queue Delta
    depth = int(max(1, ofi_cfg.depth_levels))
    decay = float(ofi_cfg.ofi_weight_decay)
    parts = []
    for i in range(1, depth + 1):
        w = decay ** (i - 1)
        bid_c = pl.when(pl.col(f"bid_p{i}") > pl.col(f"bid_p{i}").shift(1)).then(pl.col(f"bid_v{i}"))\
                  .when(pl.col(f"bid_p{i}") < pl.col(f"bid_p{i}").shift(1)).then(-pl.col(f"bid_v{i}").shift(1))\
                  .otherwise(pl.col(f"bid_v{i}") - pl.col(f"bid_v{i}").shift(1))
        ask_c = pl.when(pl.col(f"ask_p{i}") < pl.col(f"ask_p{i}").shift(1)).then(pl.col(f"ask_v{i}"))\
                  .when(pl.col(f"ask_p{i}") > pl.col(f"ask_p{i}").shift(1)).then(-pl.col(f"ask_v{i}").shift(1))\
                  .otherwise(pl.col(f"ask_v{i}") - pl.col(f"ask_v{i}").shift(1))
        parts.append((bid_c - ask_c) * w)
    return _sum_expr(parts).alias("v_ofi")

def _lob_flux_expr() -> pl.Expr:
    return (pl.col("bid_v1").diff().abs() + pl.col("ask_v1").diff().abs()).alias("lob_flux")

def build_l2_frames(path: str, cfg: L2PipelineConfig, target_frames: float | None = None) -> pl.DataFrame:
    lf = scan_l2_quotes(path, cfg)
    lf = lf.sort("time")
    
    # Filters
    s = cfg.session
    if s.enable_session_filter and not s.allow_auction:
        cond = ((pl.col("time") >= int(s.session_1_start)) & (pl.col("time") <= int(s.session_1_end))) | \
               ((pl.col("time") >= int(s.session_2_start)) & (pl.col("time") <= int(s.session_2_end)))
        lf = lf.filter(cond)

    q = cfg.quality
    if q.drop_zero_price: lf = lf.filter(pl.col("price") > float(q.min_price))
    
    # Basic features
    vc = cfg.volume_clock
    vol_expr = pl.col("vol").alias("vol_tick") if vc.volume_mode == "delta" else (pl.col("vol") - pl.col("vol").shift(1))
    vol_expr = pl.when(vol_expr > float(vc.volume_floor)).then(vol_expr).otherwise(float(vc.volume_floor)).alias("vol_tick")
    
    lf = lf.with_columns([vol_expr, _microprice_expr(cfg), _depth_expr(cfg)])
    if q.drop_nonpositive_frame_price:
        lf = lf.filter(pl.col("microprice") > float(q.min_frame_price))
        
    lf = lf.with_columns([_ofi_expr(cfg), _lob_flux_expr()])

    # --- v6.0 Fix: Causal Volume Projection ---

        pl.col("microprice").first().alias("open"),
        pl.col("microprice").last().alias("close"),
        pl.col("microprice").std().alias("sigma"),

            "srl_resid_abs": resid_abs,  # Absolute Error (for PnL calc)
            "epiplexity": epiplexity,
            "topo_area": topo_area,
            "topo_energy": topo_energy,
            "topo_winding": winding,     # Topological Invariant
            "topo_compact": compact,
            "adaptive_y": current_y_lin,
            "spoof_ratio": float(spoof),
            "is_energy_active": bool(is_active),
            # Compat keys
            "srl_resid_050": resid_z
        })
        results.append(row)

    res_df = pl.DataFrame(results)
    
    # 6. Signal Logic
    if "srl_resid" in res_df.columns:
        res_df = res_df.with_columns([
            (
                (pl.col("is_energy_active") == True) &
                (pl.col("epiplexity") > peace_threshold) &
                (pl.col("srl_resid").abs() > 1.5) & # Significant Z-score (>1.5 sigma)
                (pl.col("topo_winding").abs() > 0.1) & # Must have topological intent
                (pl.col("spoof_ratio") < float(getattr(sig, "spoofing_ratio_max", 2.5)))
            ).alias("is_signal"),
            pl.col("srl_resid").sign().alias("direction") # Contra-trade the residual
        ])
    return res_df

def apply_recursive_physics(frames, cfg, initial_y=None):
    return _apply_recursive_physics(frames, cfg, initial_y)

def run_l2_kernel(path, cfg, initial_y=None, target_frames=None):
    frames = build_l2_frames(path, cfg, target_frames)
    frames = _apply_recursive_physics(frames, cfg, initial_y)
    return frames, frames.filter(pl.col("is_signal"))

class OmegaKernel:

    def run(self, initial_y=None, target_frames=None, debug_mode=False):

        return frames

"""
physics_auditor.py
OMEGA v6.0 Auditor.
"""
from __future__ import annotations
import json, random, numpy as np, polars as pl
from pathlib import Path
from config import L2PipelineConfig
from omega_v3_core.kernel import OmegaKernel, apply_recursive_physics
from omega_v3_core.omega_math_core import calc_epiplexity

class OmegaPhysicsAuditor:
    def __init__(self, data_dir, output_dir="./model_audit", cfg=None):
        self.data_dir, self.output_dir = Path(data_dir), Path(output_dir)
        self.output_dir.mkdir(parents=True, exist_ok=True)
        self.cfg, self.files = cfg or L2PipelineConfig(), sorted(self.data_dir.glob("*.parquet"))

    def _rng(self, salt=0):
        return random.Random(int(getattr(self.cfg.epiplexity, "prior_random_seed", 42)) + salt)

    def run_continuous_calibration(self, target_frames=None, sample_frac=1.0):
        audit_data = []
        last_y = 0.75
        for f in self.files:
            k = OmegaKernel(str(f), self.cfg)
            df = k.run(initial_y=last_y, target_frames=target_frames)
            if df.is_empty(): continue
            last_y = df["adaptive_y"][-1]
            if sample_frac < 1.0: df = df.sample(fraction=sample_frac)
            cols = ["epiplexity", "srl_resid", "topo_winding", "topo_compact", "trace"]
            audit_data.append(df.select([c for c in cols if c in df.columns]))
            
        if not audit_data: return {"status": "no_data"}
        metrics = self._audit(pl.concat(audit_data))
        metrics["FINAL_Y"] = last_y
        
        with (self.output_dir / "production_config.json").open("w") as f:
            json.dump({"METRICS": metrics, "AUTO_LEARNED": {"TARGET": target_frames}}, f, indent=4)
        return metrics

    def _audit(self, df):
        traces = df["trace"].to_list()
        if not traces: return {}
        
        # Block Shuffle Null Hypothesis (Auditor Rec 3.1)
        sample = random.sample(traces, min(len(traces), 500))
        real = [calc_epiplexity(t, self.cfg.epiplexity) for t in sample]
        null = []
        rng = np.random.default_rng(42)
        block_size = 8
        
        for t in sample:
            arr = np.array(t)
            if len(arr) > block_size * 2:
                n_blks = len(arr) // block_size
                blocks = np.array_split(arr[:n_blks*block_size], n_blks)
                rng.shuffle(blocks)
                null.append(calc_epiplexity(np.concatenate(blocks), self.cfg.epiplexity))
        
        # Robust SNR (Auditor Rec 3.2)
        null_med = np.median(null) if null else 0.0
        null_mad = np.median(np.abs(null - null_med)) + 1e-9
        snr = (np.median(real) - null_med) / null_mad
        
        ortho = 0.0
        if "srl_resid" in df.columns:
             ortho = np.corrcoef(df["epiplexity"], df["srl_resid"].abs())[0,1]

        return {"Topo_SNR_Robust": float(snr), "Orthogonality": float(ortho)}
        
    def derive_market_priors(self): return {}
    def run_renormalization_scan(self): return self.run_continuous_calibration()

"""
trainer.py
Level-2 (v6.0) Trainer.
"""
from __future__ import annotations
import pickle, os, numpy as np, polars as pl
from pathlib import Path
from sklearn.linear_model import SGDClassifier
from sklearn.preprocessing import StandardScaler
from config import L2PipelineConfig
from omega_v3_core.kernel import apply_recursive_physics
from tools.multi_dir_loader import discover_l2_dirs

class OmegaTrainerV3:
    def __init__(self, cfg: L2PipelineConfig):
        self.cfg, self.scaler = cfg, StandardScaler()
        self.model = SGDClassifier(loss="log_loss", penalty="l2", alpha=1e-4, average=True)
        self.feature_cols = [
            "sigma_eff", "net_ofi", "depth_eff", 
            "epiplexity",    # Robust Spectral Gain
            "srl_resid",     # Dimensionless Force (Z-Score)
            "topo_area", "topo_energy", "topo_winding", "topo_compact", # Advanced Topology
            "price_change", "bar_duration_ms", "adaptive_y"
        ]
        self.label_col = "direction_label"

    def train(self, sample_frac=1.0):
        print("Starting training OMEGA v6.0 (The Invariant Observer)...")
        dirs = discover_l2_dirs()
        if not dirs: return
        
        classes = np.array([-1, 0, 1])
        if self.cfg.train.drop_neutral_labels: classes = np.array([-1, 1])
        
        total = 0
        for d in dirs:
            for f in sorted(list(d.glob("*.parquet"))):
                try:
                    lf = pl.scan_parquet(str(f))
                    if sample_frac < 1.0: lf = lf.sample(fraction=sample_frac)
                    df = lf.collect()
                    if df.height == 0: continue
                    
                    # Physics v6
                    df = apply_recursive_physics(df, self.cfg)
                    
                    # Labels
                    tcfg = cfg = self.cfg.train
                    horizon = int(tcfg.label_horizon_buckets)
                    df = df.with_columns([
                        (pl.col("close").shift(-horizon) - pl.col("close")).alias("fwd_change")
                    ]).with_columns([
                        (pl.col("fwd_change") / pl.col("close")).alias("ret_k"),
                        (pl.col("sigma_eff") / pl.col("close")).alias("sigma_ret")
                    ]).with_columns(
                        pl.when(pl.col("ret_k").abs() > float(tcfg.label_sigma_mult) * pl.col("sigma_ret"))
                        .then(pl.col("ret_k").sign()).otherwise(0.0).alias(self.label_col)
                    ).drop_nulls()
                    
                    missing = [c for c in self.feature_cols if c not in df.columns]
                    if missing: continue

                    X = df.select(self.feature_cols).to_numpy()
                    y = df.select(self.label_col).to_numpy().ravel()
                    
                    # Structure Weighting: Trust high epiplexity samples more
                    w = np.clip(df["epiplexity"].to_numpy(), 0.1, 2.0)
                    
                    self.scaler.partial_fit(X)
                    X_sc = self.scaler.transform(X)
                    self.model.partial_fit(X_sc, y, classes=classes, sample_weight=w)
                    
                    total += len(y)
                    if total % 200000 == 0: print(f"Rows: {total}")
                except Exception as e: print(f"Err {f.name}: {e}")
        self.save()

    def save(self):
        p = Path("./artifacts/omega_v6_model.pkl")
        p.parent.mkdir(parents=True, exist_ok=True)
        with open(p, "wb") as f:
            pickle.dump({"model": self.model, "scaler": self.scaler, "features": self.feature_cols}, f)
        print(f"Saved to {p}")
